La palabra reservada que se utiliza dentro de una función para indicar que ésta no devuelve valor alguno, o para indicar que no recibe parámetros es:
- VOID


Determine cuál es la función del & en la función restar:
int restar(int*, int, int);

int main(void)
{
	int primerNum;
	int segundoNum;
	int resultado;

	printf("Ingrese numero 1: ");
	scanf("%d", &primerNum);

	printf("Ingrese numero 2: ");
	scanf("%d", &segundoNum);

	restar(&resultado, primerNum, segundoNum);
	printf("El resultado es: %d", resultado);
	return 0;
}

int restar(int* resultado, int num1, int num2)
{
	*resultado = num1 -num2;
	
	return 0;
